한 개의 정수를 입력받아서 1부터 입력받은 정수까지의 홀수합을 출력하시오.
```java
import java.util.Scanner; // 키보드 입력을 위한 Scanner 클래스 불러오기

public class Problem03 {
    public static void main(String[] args) {
        // Scanner 객체 생성
        Scanner scanner = new Scanner(System.in);
        
        // 사용자에게 정수 입력 요청
        System.out.print("정수 입력: ");
        int num = scanner.nextInt(); // 정수 입력 받기
        
        int sum = 0; // 홀수 합을 저장할 변수 초기화
        
        // 1부터 입력받은 정수까지 반복
        for (int i = 1; i <= num; i++) {
            // 홀수인지 확인 (2로 나누었을 때 나머지가 1이면 홀수)
            if (i % 2 == 1) {
                sum += i; // 홀수면 합계에 더함
            }
        }
        
        // 결과 출력
        System.out.println("1부터 " + num + "까지 홀수 합: " + sum);
        
        // Scanner 자원 해제
        scanner.close();
    }
}
```
