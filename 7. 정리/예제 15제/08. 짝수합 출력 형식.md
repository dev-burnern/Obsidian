1부터 10까지 짝수합을 다음과 같은 형태로 출력하시오.(for문, if문 사용)  
출력 형태: 2+4+6+8+10=30
```java
public class Problem08 {
    public static void main(String[] args) {
        int sum = 0; // 합계를 저장할 변수 초기화
        
        // 결과 출력을 위한 문자열 변수
        String output = "";
        
        // 1부터 10까지 반복
        for (int i = 1; i <= 10; i++) {
            // 짝수인지 확인 (2로 나누어 떨어지면 짝수)
            if (i % 2 == 0) {
                sum += i; // 짝수면 합계에 더함
                
                // 출력 형태 구성
                output += i;
                
                // 마지막 숫자가 아니면 '+' 추가
                if (i != 10) {
                    output += "+";
                }
            }
        }
        
        // 최종 결과 출력
        System.out.println(output + "=" + sum);
    }
}
```
