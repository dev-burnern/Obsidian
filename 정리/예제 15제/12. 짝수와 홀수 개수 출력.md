-1이 입력될때 까지 정수를 입력받고 짝수와 홀수의 개수를 출력하시오.
```java
import java.util.Scanner; // 키보드 입력을 위한 Scanner 클래스 불러오기

public class Problem12 {
    public static void main(String[] args) {
        // Scanner 객체 생성
        Scanner scanner = new Scanner(System.in);
        
        int evenCount = 0; // 짝수 개수를 저장할 변수
        int oddCount = 0;  // 홀수 개수를 저장할 변수
        int num;           // 사용자 입력을 저장할 변수
        
        // -1이 입력될 때까지 반복
        while (true) {
            System.out.print("정수 입력(종료:-1): ");
            num = scanner.nextInt(); // 정수 입력 받기
            
            // -1이 입력되면 반복 종료
            if (num == -1) {
                break;
            }
            
            // 짝수/홀수 판단 및 개수 증가
            if (num % 2 == 0) {
                evenCount++; // 짝수 개수 증가
            } else {
                oddCount++;  // 홀수 개수 증가
            }
        }
        
        // 결과 출력
        System.out.println("짝수: " + evenCount + "개, 홀수: " + oddCount + 
                          "개, 전체: " + (evenCount + oddCount) + "개 입력");
        
        // Scanner 자원 해제
        scanner.close();
    }
}
```
