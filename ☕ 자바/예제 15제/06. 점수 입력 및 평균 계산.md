사용자로부터 점수를 입력받고 총점과 평균을 계산하여 출력하는 프로그램을 작성하시오. 단, 음수가 입력되면 점수 입력을 종료하고 계산결과를 출력한다.
```java
import java.util.Scanner; // 키보드 입력을 위한 Scanner 클래스 불러오기

public class Problem06 {
    public static void main(String[] args) {
        // Scanner 객체 생성
        Scanner scanner = new Scanner(System.in);
        
        int sum = 0;     // 총점을 저장할 변수 초기화
        int count = 0;   // 입력받은 점수의 개수를 저장할 변수 초기화
        
        while (true) {  // 무한 반복
            // 사용자에게 점수 입력 요청
            System.out.print("점수 입력(종료:-1): ");
            int score = scanner.nextInt(); // 점수 입력 받기
            
            // 음수가 입력되면 반복 종료
            if (score < 0) {
                break;
            }
            
            sum += score;  // 점수를 총점에 더함
            count++;       // 점수 개수 증가
        }
        
        // 입력받은 점수가 없는 경우
        if (count == 0) {
            System.out.println("입력된 점수가 없습니다.");
        } else {
            // 평균 계산 (double로 형변환하여 소수점 결과 얻기)
            double average = (double) sum / count;
            
            // 결과 출력
            System.out.println("총점: " + sum + ", 입력 횟수: " + count + ", 평균: " + String.format("%.2f", average));
        }
        
        // Scanner 자원 해제
        scanner.close();
    }
}
```
