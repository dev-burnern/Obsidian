사용자가 제시된 메뉴중 한개를 선택하여 선택된 번호를 표시하는 프로그램을 작성하시오. 조건: 4번 종료를 선택하면 종료하도록 하고, 잘못된 번호를 선택하면 잘못 선택했음을 표시하고 재 입력 받도록 한다.(do-while문 사용)
```java
import java.util.Scanner; // 키보드 입력을 위한 Scanner 클래스 불러오기

public class Problem07 {
    public static void main(String[] args) {
        // Scanner 객체 생성
        Scanner scanner = new Scanner(System.in);
        
        int choice; // 사용자 선택을 저장할 변수
        
        do {
            // 메뉴 출력
            System.out.println("1. 사각형");
            System.out.println("2. 원");
            System.out.println("3. 삼각형");
            System.out.println("4. 종료");
            System.out.print("메뉴를 선택하시오: ");
            
            choice = scanner.nextInt(); // 사용자 선택 입력 받기
            
            // 선택에 따른 처리
            switch (choice) {
                case 1:
                    System.out.println("사각형을 선택했습니다.");
                    break;
                case 2:
                    System.out.println("원을 선택했습니다.");
                    break;
                case 3:
                    System.out.println("삼각형을 선택했습니다.");
                    break;
                case 4:
                    System.out.println("프로그램을 종료합니다.");
                    break;
                default:
                    System.out.println("잘못 선택했습니다.");
                    break;
            }
            
            System.out.println(); // 빈 줄 출력
            
        } while (choice != 4); // 4를 선택하지 않으면 반복
        
        // Scanner 자원 해제
        scanner.close();
    }
}
```
