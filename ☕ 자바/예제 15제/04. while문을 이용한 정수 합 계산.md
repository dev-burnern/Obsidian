정수를 입력 받고 1부터 입력 받은 정수까지의 합을 구하는 프로그램을 작성하시오.(while문을 이용)
```java
import java.util.Scanner; // 키보드 입력을 위한 Scanner 클래스 불러오기

public class Problem04 {
    public static void main(String[] args) {
        // Scanner 객체 생성
        Scanner scanner = new Scanner(System.in);
        
        // 사용자에게 정수 입력 요청
        System.out.print("정수 입력: ");
        int num = scanner.nextInt(); // 정수 입력 받기
        
        int sum = 0; // 합계를 저장할 변수 초기화
        int i = 1;   // 카운터 변수 초기화
        
        // 1부터 입력받은 정수까지 반복 (while문 사용)
        while (i <= num) {
            sum += i; // 현재 숫자를 합계에 더함
            i++;      // 카운터 증가
        }
        
        // 결과 출력
        System.out.println("1부터 " + num + "까지 합: " + sum);
        
        // Scanner 자원 해제
        scanner.close();
    }
}

```
