public class q15 {  
public static void main(String[] args) {  
		for (int i = 0; i < 10; i++) {  
		// 왼쪽 별  
			System.out.print("*".repeat(9 - i));  
			// 공백(첫 줄만 제외)  
			if (i > 0) System.out.print(" ");  
			// 오른쪽 별  
			System.out.print("*".repeat(i));  
			// 줄 바꿈  
			System.out.println();  
}  
}
