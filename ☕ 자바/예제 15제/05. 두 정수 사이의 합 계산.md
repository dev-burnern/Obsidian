두 개의 정수를 입력 받고 입력받은 정수중 작은수부터 큰수까지의 합을 구하는 프로그램을 작성하시오.(for문, if문을 이용)
```java
import java.util.Scanner; // 키보드 입력을 위한 Scanner 클래스 불러오기

public class Problem05 {
    public static void main(String[] args) {
        // Scanner 객체 생성
        Scanner scanner = new Scanner(System.in);
        
        // 사용자에게 두 정수 입력 요청
        System.out.print("두 개의 정수 입력: ");
        int num1 = scanner.nextInt(); // 첫 번째 정수 입력
        int num2 = scanner.nextInt(); // 두 번째 정수 입력
        
        int small, large; // 작은 수와 큰 수를 저장할 변수
        
        // 두 수 중 작은 수와 큰 수 결정 (if문 사용)
        if (num1 < num2) {
            small = num1;
            large = num2;
        } else {
            small = num2;
            large = num1;
        }
        
        int sum = 0; // 합계를 저장할 변수 초기화
        
        // 작은 수부터 큰 수까지 반복 (for문 사용)
        for (int i = small; i <= large; i++) {
            sum += i; // 현재 숫자를 합계에 더함
        }
        
        // 결과 출력
        System.out.println(small + "부터 " + large + "까지의 합: " + sum);
        
        // Scanner 자원 해제
        scanner.close();
    }
}
```
