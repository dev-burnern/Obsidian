1부터 10까지 홀수합을 다음과 같은 형태로 출력하시오.(for문, if문 사용)  
출력 형태: 1+3+5+7+9=25
```java
public class Problem09 {
    public static void main(String[] args) {
        int sum = 0; // 합계를 저장할 변수 초기화
        
        // 결과 출력을 위한 문자열 변수
        String output = "";
        
        // 1부터 10까지 반복
        for (int i = 1; i <= 10; i++) {
            // 홀수인지 확인 (2로 나눈 나머지가 1이면 홀수)
            if (i % 2 == 1) {
                sum += i; // 홀수면 합계에 더함
                
                // 출력 형태 구성
                output += i;
                
                // 마지막 홀수가 아니면 '+' 추가
                if (i != 9) {
                    output += "+";
                }
            }
        }
        
        // 최종 결과 출력
        System.out.println(output + "=" + sum);
    }
}
```
